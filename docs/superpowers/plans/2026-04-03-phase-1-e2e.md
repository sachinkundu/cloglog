```python
"""End-to-end tests for agent registration, task lifecycle, and status roll-up."""

import pytest
from httpx import AsyncClient


def _auth(api_key: str) -> dict:
    return {"Authorization": f"Bearer {api_key}"}


@pytest.fixture
async def project_with_tasks(client: AsyncClient):
    """Create a project with imported tasks, return IDs + api_key."""
    p = (await client.post("/api/v1/projects", json={"name": "e2e-agent"})).json()
    await client.post(f"/api/v1/projects/{p['id']}/import", json={
        "epics": [{
            "title": "Epic",
            "features": [{
                "title": "Feature",
                "tasks": [
                    {"title": "Task 1"},
                    {"title": "Task 2"},
                    {"title": "Task 3"},
                ],
            }],
        }],
    })
    board = (await client.get(f"/api/v1/projects/{p['id']}/board")).json()
    backlog = next(c for c in board["columns"] if c["status"] == "backlog")
    task_ids = [t["id"] for t in backlog["tasks"]]
    return {
        "project_id": p["id"],
        "api_key": p["api_key"],
        "task_ids": task_ids,
    }


async def test_agent_register_and_resume(client: AsyncClient, project_with_tasks: dict):
    """Agent registers, unregisters, and re-registers (resumes)."""
    h = _auth(project_with_tasks["api_key"])

    # First registration
    reg1 = (await client.post(
        "/api/v1/agents/register",
        json={"worktree_path": "/home/user/wt-auth"},
        headers=h,
    )).json()
    assert reg1["name"] == "wt-auth"
    assert reg1["resumed"] is False
    wt_id = reg1["worktree_id"]

    # Unregister
    unreg = await client.post(f"/api/v1/agents/{wt_id}/unregister", headers=h)
    assert unreg.status_code == 200

    # Re-register (should resume)
    reg2 = (await client.post(
        "/api/v1/agents/register",
        json={"worktree_path": "/home/user/wt-auth"},
        headers=h,
    )).json()
    assert reg2["resumed"] is True
    assert reg2["worktree_id"] == wt_id


async def test_agent_task_lifecycle(client: AsyncClient, project_with_tasks: dict):
    """Agent starts a task, completes it, board reflects changes."""
    h = _auth(project_with_tasks["api_key"])
    project_id = project_with_tasks["project_id"]
    task_id = project_with_tasks["task_ids"][0]

    # Register agent
    reg = (await client.post(
        "/api/v1/agents/register",
        json={"worktree_path": "/home/user/wt-lifecycle"},
        headers=h,
    )).json()
    wt_id = reg["worktree_id"]

    # Assign task to worktree
    await client.patch(f"/api/v1/tasks/{task_id}", json={"worktree_id": wt_id})

    # Start task
    start_resp = await client.post(
        f"/api/v1/agents/{wt_id}/start-task",
        json={"task_id": task_id},
        headers=h,
    )
    assert start_resp.status_code == 200

    # Verify board shows task in_progress
    board = (await client.get(f"/api/v1/projects/{project_id}/board")).json()
    in_progress = next(c for c in board["columns"] if c["status"] == "in_progress")
    assert any(t["id"] == task_id for t in in_progress["tasks"])

    # Complete task
    complete_resp = await client.post(
        f"/api/v1/agents/{wt_id}/complete-task",
        json={"task_id": task_id},
        headers=h,
    )
    assert complete_resp.status_code == 200

    # Verify board shows task done
    board = (await client.get(f"/api/v1/projects/{project_id}/board")).json()
    done = next(c for c in board["columns"] if c["status"] == "done")
    assert any(t["id"] == task_id for t in done["tasks"])
    assert board["done_count"] == 1


async def test_agent_status_rollup(client: AsyncClient, project_with_tasks: dict):
    """Completing all tasks rolls up feature and epic to done."""
    h = _auth(project_with_tasks["api_key"])
    project_id = project_with_tasks["project_id"]

    # Register
    reg = (await client.post(
        "/api/v1/agents/register",
        json={"worktree_path": "/home/user/wt-rollup"},

    # Complete all 3 tasks

    assert board["total_tasks"] == 3

    # Verify epic rolled up
    epics = (await client.get(f"/api/v1/projects/{project_id}/epics")).json()
```

```python
# tests/e2e/test_full_workflow.py
"""End-to-end test: complete agent workflow from project creation to task completion."""

import pytest
from httpx import AsyncClient


def _auth(api_key: str) -> dict:
    return {"Authorization": f"Bearer {api_key}"}


async def test_full_agent_workflow(client: AsyncClient):
    """
    Complete workflow:
    1. Create project (get API key)
    2. Import plan (epics/features/tasks)
    3. Verify board shows tasks in backlog
    4. Agent registers
    5. Agent starts a task
    6. Agent attaches a document
    7. Agent completes the task
    8. Verify board reflects all changes
    9. Verify document is retrievable
    10. Agent unregisters
    """
    # 1. Create project
    project = (await client.post(
        "/api/v1/projects", json={"name": "e2e-full-workflow"}
    )).json()
    project_id = project["id"]
    api_key = project["api_key"]
    h = _auth(api_key)

    # 2. Import plan
    plan = {
        "epics": [{
            "title": "Core",
            "features": [{
                "title": "User Auth",
                "tasks": [
                    {"title": "Login endpoint"},
                    {"title": "Token validation"},
                ],
            }],
        }],
    }
    import_resp = await client.post(f"/api/v1/projects/{project_id}/import", json=plan)
    assert import_resp.status_code == 201

    # 3. Verify board
    board = (await client.get(f"/api/v1/projects/{project_id}/board")).json()
    assert board["total_tasks"] == 2
    assert board["done_count"] == 0
    backlog = next(c for c in board["columns"] if c["status"] == "backlog")
    task_id = backlog["tasks"][0]["id"]

    # 4. Agent registers
    reg = (await client.post(
        "/api/v1/agents/register",
        json={"worktree_path": "/home/user/wt-core"},
        headers=h,
    )).json()
    wt_id = reg["worktree_id"]
    assert reg["name"] == "wt-core"

    # Assign task
    await client.patch(f"/api/v1/tasks/{task_id}", json={"worktree_id": wt_id})

    # 5. Agent starts task
    await client.post(
        f"/api/v1/agents/{wt_id}/start-task",
        json={"task_id": task_id}, headers=h,
    )

    # Verify in_progress
    board = (await client.get(f"/api/v1/projects/{project_id}/board")).json()
    in_progress = next(c for c in board["columns"] if c["status"] == "in_progress")
    assert any(t["id"] == task_id for t in in_progress["tasks"])

    # 6. Agent attaches a document

        f"/api/v1/agents/{wt_id}/documents",

            "task_id": task_id,
            "type": "spec",
            "title": "Login Spec",
            "content": "# Login\n\nPOST /auth/login with email+password returns JWT.",
        },
        headers=h,
    )).json()
    assert doc["title"] == "Login Spec"

    # 7. Agent completes task
    complete = (await client.post(
        f"/api/v1/agents/{wt_id}/complete-task",

    # 9. Verify document
```

```python
# tests/e2e/test_auth.py
"""End-to-end auth tests: API key validation across all protected endpoints."""

import pytest
from httpx import AsyncClient


async def test_agent_endpoints_require_auth(client: AsyncClient):
    """All agent-facing endpoints reject unauthenticated requests."""
    endpoints = [
        ("POST", "/api/v1/agents/register", {"worktree_path": "/tmp"}),
        ("POST", "/api/v1/agents/00000000-0000-0000-0000-000000000000/heartbeat", None),
        ("POST", "/api/v1/agents/00000000-0000-0000-0000-000000000000/start-task", {"task_id": "00000000-0000-0000-0000-000000000000"}),
        ("POST", "/api/v1/agents/00000000-0000-0000-0000-000000000000/complete-task", {"task_id": "00000000-0000-0000-0000-000000000000"}),
        ("POST", "/api/v1/agents/00000000-0000-0000-0000-000000000000/unregister", None),
        ("POST", "/api/v1/agents/00000000-0000-0000-0000-000000000000/documents", {"task_id": "00000000-0000-0000-0000-000000000000", "type": "spec", "title": "t", "content": "c"}),
    ]

    for method, path, body in endpoints:
        if method == "POST":
            resp = await client.post(path, json=body)
        else:
            resp = await client.request(method, path)
        assert resp.status_code == 401, f"{method} {path} should require auth, got {resp.status_code}"


async def test_invalid_api_key_rejected(client: AsyncClient):
    """An invalid API key is rejected on agent endpoints."""
    bad_headers = {"Authorization": "Bearer not-a-real-key"}
    resp = await client.post(
        "/api/v1/agents/register",
        json={"worktree_path": "/tmp"},
        headers=bad_headers,
    )
    assert resp.status_code == 401


async def test_dashboard_endpoints_are_public(client: AsyncClient):
    """Dashboard-read endpoints don't require auth."""
    # These should return 200 (or 404 for missing resources), not 401
    resp = await client.get("/api/v1/projects")
    assert resp.status_code == 200

    resp = await client.get("/api/v1/documents/00000000-0000-0000-0000-000000000000")
    assert resp.status_code == 404  # Not found, but not 401
```
